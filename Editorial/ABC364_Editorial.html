B - Grid Walk

statement

有一个网格，网格中有 HHH 行和 WWW 列。让 (i,j)(i, j)(i,j) 表示从上往下 iii 行，从左往上 jjj 列的单元格。

如果 Ci,jC_{i, j}Ci,j​ 是 .，那么单元格 (i,j)(i, j)(i,j) 就是空的，如果 Ci,jC_{i, j}Ci,j​ 是 #，那么单元格 (i,j)(i, j)(i,j) 就不是空的。

高桥目前位于 (Si,Sj)(S_i, S_j)(Si​,Sj​) 单元格，他将按照以下规则依次对 i=1,2,…,∣X∣i = 1, 2, \ldots, |X|i=1,2,…,∣X∣ 采取行动。

  • XiX_iXi​ 是 L，并且当前单元格左边的单元格存在且是空的，那么他将移动到左边的单元格。否则，他将留在当前的单元格中。
  • XiX_iXi​ 是 R，并且当前单元格右边的单元格存在且是空的，那么他将移动到右边的单元格。否则，他将留在当前的单元格中。
  • XiX_iXi​ 是 U，并且当前单元格上方的单元格存在且是空的，那么他将移动到上方的单元格。否则，他将留在当前的单元格中。
  • XiX_iXi​ 是 D，并且当前单元格下方的单元格存在且是空的，那么他将移动到下方的单元格。否则，他将留在当前的单元格中。

C - Minimum Glutton

statement

有 NNN 道菜，其中 iii 道菜的甜度为 AiA_iAi​ ，咸度为 BiB_iBi​ 。

高桥打算将这些 NNN 菜肴按照自己喜欢的顺序排列，然后按照这个顺序吃掉。

他将按照排列顺序吃掉这些菜肴，但是一旦他吃掉的菜肴的总甜度超过 XXX 或总咸度超过 YYY ，他就会停止进食。

求他最后吃掉的菜肴的最少数量。

analysis

由于这题只需要满足甜度或咸度大于给定值将会退出进食模式，那么这里直接分两种情况讨论：由于甜度过大退出，由于咸度过大退出。分别计算出这两种情况下至少需要吃掉多少道菜，然后取最小值输出即可。注意开long long。

code

#include<bits/stdc++.h>
#define pii pair<int, int>
#define endl "\n"
#define setpre(x) fixed << setprecision(x)
#define FOR(i, l, r) for(int i = l; i <= r; ++i)
#define ROF(i, l, r) for(int i = r; i >= l; --i)
#define fi first

constexpr int MAXN = 5e5 + 10;
constexpr int inf = 0x3f3f3f3f;
constexpr LL  INF = 0x3f3f3f3fffffffff;

void solve() {
    int n;
    LL x, y;
    cin >> n >> x >> y;
    vector<int> a(n), b(n);
    for(auto& v : a) cin >> v;
    for(auto& v : b) cin >> v;
    sort(a.rbegin(), a.rend());
    sort(b.rbegin(), b.rend());
    LL res1 = 0, now1 = 0;
    for(auto t : a) {

        if(now1 > x) break;
    }

    for(auto t : b) {
        now2 += t, res2++;
        if(now2 > y) break;
    }
    cout << min(res1, res2) << endl;
}

    cin.tie(0), cout.tie(0) -> sync_with_stdio(0);
    int t = 1;
    // cin >> t;

    return 0;
}

D - K-th Nearest

statement

在一条数线上有 NNN 个点 A1,…,ANA_1,\dots,A_NA1​,…,AN​ ，其中点 AiA_iAi​ 的坐标为 aia_iai​。

就每个 j=1,2,…,Qj=1,2,\dots,Qj=1,2,…,Q 回答下面的问题：

  • 给一个点 BBB，坐标为 bbb 和一个正整数 kkk
  • 设 XXX 是 A1,A2,…,ANA_1,A_2,\dots,A_NA1​,A2​,…,AN​ 中离点 BBB 第 kkk 近的点。求点 XXX 与 BBB 之间的距离。更具体地说，设 did_idi​ 是点 AiA_iAi​ 与 BBB 之间的距离。将 (d1,d2,…,dN)(d_1,d_2,\dots,d_N)(d1​,d2​,…,dN​) 按升序排序，得到序列 (d1′,d2′,…,dN′)(d_1',d_2',\dots,d_N')(d1′​,d2′​,…,dN′​) 。求 dk′d_{k}'dk′​ .

hint

  • 1≤N,Q≤1051 \leq N, Q \leq 10^51≤N,Q≤105
  • −108≤ai,bj≤108-10^8 \leq a_i, b_j \leq 10^8−108≤ai​,bj​≤108
  • 1≤kj≤N1 \leq k_j \leq N1≤kj​≤N
  • 所有输入值均为整数。

analysis

这里首先先将问题转化：求最小的 ddd，使得区间 [x−d,x+d][x - d, x + d][x−d,x+d] 内所含的点的个数至少为 kkk，容易看出这是一个具有答案单调性的问题，所以使用二分答案解决。

vector<vector<int>> dp(x + 1, vector<int>(y + 1));
for(int i = 1; i <= n; ++i) {

F - Range Connect MST

statement

有一个图，它有 N+QN + QN+Q 个顶点，编号为 1,2,…,N+Q1, 2, \ldots, N + Q1,2,…,N+Q 。最初，该图没有边。

对于这个图，请依次对 i=1,2,…,Qi = 1, 2, \ldots, Qi=1,2,…,Q 执行以下操作：

  • 对于每个满足 Li≤j≤RiL_i \leq j \leq R_iLi​≤j≤Ri​ 的整数 jjj ，在顶点 N+iN + iN+i 和 jjj 之间添加一条代价为 CiC_iCi​ 的无向边。

完成所有操作后，确定图形是否相连。如果相连，求该图的最小生成树的代价。

最小生成树是成本最小的生成树，生成树的成本是生成树所用边的成本之和。

hint

  • 1≤N,Q≤2×1051 \leq N, Q \leq 2 \times 10^51≤N,Q≤2×105
  • 1≤Li≤Ri≤N1 \leq L_i \leq R_i \leq N1≤Li​≤Ri​≤N
  • 1≤Ci≤1091 \leq C_i \leq 10^91≤Ci​≤109
  • 所有输入值均为整数。

analysis

既然要求最小生成树，毫无疑问就是用 Kruskal\text{Kruskal}Kruskal 算法，先将所有的连边方式按照代价升序排列。先考虑暴力解法，考虑第 jjj 次操作，该操作由三元组描述 [L,R,C][L, R, C][L,R,C]，按照 Kruskal\text{Kruskal}Kruskal 的操作顺序，需要依次判断 [L,R][L, R][L,R] 中的点与 N+jN + jN+j 的联通性，时间复杂度 O(N⋅Q⋅α(N))O(N\cdot Q\cdot \alpha (N))O(N⋅Q⋅α(N))。下面考虑优化。

将区间 [L,R][L, R][L,R] 中的点和 N+jN + jN+j 连接这个操作有一个重要的性质，如果注意到该性质才能继续往下做：并查集中的 pre\text{pre}pre 数组单调不减，例如 pre={1,1,3,4,4,5}\text{pre} = \{1, 1, 3, 4, 4, 5\}pre={1,1,3,4,4,5}

证明其实非常简单，数学归纳法yy一下就出来了。

进而我们会发现，pre\text{pre}pre 其实是由一块块数字组成的，所以我们在区间 [Lj,Rj][L_j, R_j][Lj​,Rj​] 中将所有的块连接起来即可，接下来就是如何快速找到所有的块。而在并查集中，连通块内任意一点作为根都是可以的，所以我们不妨令根为这个区间的最后一个点。也就是说 ∀  i  (pre[i]=j),  i≤j\forall\;i\;(\text{pre}[i] = j),\; i\le j∀i(pre[i]=j),i≤j.

那么我们只需要 merge(x + 1, x) 然后 x = find(x) 即可.

code

#include<bits/stdc++.h>
#define pii pair<int, int>
#define endl "\n"
#define setpre(x) fixed << setprecision(x)
#define FOR(i, l, r) for(int i = l; i <= r; ++i)
#define ROF(i, l, r) for(int i = r; i >= l; --i)
#define fi first
#define se second
using namespace std;
using LL = long long;
constexpr int MAXN = 5e5 + 10;
constexpr int inf = 0x3f3f3f3f;
constexpr LL  INF = 0x3f3f3f3fffffffff;

struct DSU {
    int n;
    vector<int> pre, siz;
    DSU (int n) : n(n) {
        pre.resize(n + 1);
        siz.resize(n + 1, 1);
        iota(pre.begin(), pre.end(), 0);
    }

    int find(int x) {
        if(pre[x] == x) return x;
        return pre[x] = find(pre[x]);
    }

    // 后树合并到前树
    void merge(int x, int y) {
        x = find(x), y = find(y);
        if(x ^ y) {
            pre[y] = x;
            siz[x] += siz[y];
        }
    }
    int size(int x) { return siz[find(x)]; }
};

void solve() {
    int n, q;
    cin >> n >> q;
    vector<array<int, 3>> a(q + 1);
    FOR(i, 1, q) cin >> a[i][1] >> a[i][2] >> a[i][0];
    sort(a.begin() + 1, a.end());

    DSU dsu(n);
    LL ans = 0;
    for(int i = 1; i <= q; ++i) {
        LL wei = a[i][0];
        int l = a[i][1], r = a[i][2];
        ans += wei;
        for(int x = dsu.find(l); x + 1 <= r; x = dsu.find(x)) {
            dsu.merge(x + 1, x);
            ans += wei;
        }
    }
    if(dsu.size(1) == n) cout << ans << endl;
    else cout << -1 << endl;
}

signed main() {
    cin.tie(0), cout.tie(0) -> sync_with_stdio(0);
    int t = 1;
    // cin >> t;
    while(t--) solve();
    return 0;
}

analysis

这题其实还有一个解法。由于这是一个对区间进行操作的题目，不难想到使用一些巧妙数据结构来做。

通过上面的分析，我们知道区间中每有一种数字，都要连一条边，这就很自然地产生这个想法：我们用线段树维护 pre\text{pre}pre 数组区间中出现的数字的种数。

根据前文关于 pre\text{pre}pre 单调性的描述可以知道，相邻两个区间如果出现了相同的数字，那么必定是交界处两个数字相同。所以再维护一下区间左右端点的数字即可，合并的函数就这么写：

auto merge = [&](node l, node r, node u) -> node {
    u.l = l.l, u.r = r.r;
    u.prel = l.prel, u.prer = r.prer;
    u.ans = l.ans + r.ans - (l.prer == r.prel);
    return u;
};

由于合并后整个区间处于同一个连通块内，我们不难想到将整个区间设置为同一个数，这样更新的方式也有了。但是这样有一个问题：不能只更新 [Lj,Rj][L_j, R_j][Lj​,Rj​]。例如 pre={1,1,1,2,2}, L=3, R=4\text{pre} = \{1, 1, 1, 2, 2\},\,L = 3,\,R = 4pre={1,1,1,2,2},L=3,R=4 这个时候无论将 [3,  4][3,\;4][3,4] 更新成 111 还是 222 都会导致整个数组中含有两种数字。

解决方法也很简单，只需要将更新的区间扩大一下就好了，即将上面的 [3,  4][3,\;4][3,4] 改为 [1,  5][1,\;5][1,5]。开两个数组 left [i],  right [i]\text{left}\,[i], \;\text{right}\,[i]left[i],right[i] 记录数字 iii 出现区间的左端点和右端点，然后更新区间 [  left [prel],  right [prer]  ][\;\text{left}\,[prel], \;\text{right}\,[prer]\;][left[prel],right[prer]] 即可。

code

#include<bits/stdc++.h>
#define pii pair<int, int>
#define endl "\n"
#define setpre(x) fixed << setprecision(x)
#define FOR(i, l, r) for(int i = l; i <= r; ++i)
#define ROF(i, l, r) for(int i = r; i >= l; --i)
#define fi first
#define se second
using namespace std;
using LL = long long;
constexpr int MAXN = 5e5 + 10;
constexpr int inf = 0x3f3f3f3f;
constexpr LL  INF = 0x3f3f3f3fffffffff;

struct node{
    int l, r, tag;
    int prel, prer, ans;
};
#define ls (u << 1)
#define rs (u << 1 | 1)

void solve() {
    int n, q;
    cin >> n >> q;
    vector<array<int, 3>> a(q + 1);
    FOR(i, 1, q) cin >> a[i][1] >> a[i][2] >> a[i][0];
    sort(a.begin() + 1, a.end());
    
    vector<int> left(n + 1), right(n + 1);
    FOR(i, 1, n) left[i] = right[i] = i;

    vector<node> tr(n << 2);
    auto merge = [&](node l, node r, node u = {}) -> node {
        u.l = l.l, u.r = r.r;
        u.prel = l.prel, u.prer = r.prer;
        u.ans = l.ans + r.ans - (l.prer == r.prel);
        return u;
    };
    auto pushup = [&](int u) -> void {
        tr[u] = merge(tr[ls], tr[rs], tr[u]);
    };
    auto pushdown = [&](int u) -> void {
        if(tr[u].tag) {
            tr[ls].prel = tr[ls].prer = tr[rs].prel = tr[rs].prer = tr[u].tag;
            tr[ls].tag = tr[rs].tag = tr[u].tag;
            tr[ls].ans = tr[rs].ans = 1;
            tr[u].tag = 0;
        }
    };
    auto build = [&](auto build, int l, int r, int u) -> void {
        tr[u].l = l, tr[u].r = r;
        if(l == r) {
            tr[u].prel = tr[u].prer = l;
            tr[u].tag = 0;
            tr[u].ans = 1;
            return;
        }
        int m = l + r >> 1;
        build(build, l, m, ls);
        build(build, m + 1, r, rs);
        pushup(u);
    };
    auto modify = [&](auto modify, int l, int r, int u, int val) -> void {
        if(l <= tr[u].l && tr[u].r <= r) {
            tr[u].ans = 1;
            tr[u].prel = tr[u].prer = tr[u].tag = val;
            return;
        }
        pushdown(u);
        if(l <= tr[ls].r) modify(modify, l, r, ls, val);
        if(r >= tr[rs].l) modify(modify, l, r, rs, val);
        pushup(u);
    };
    auto query = [&](auto query, int l, int r, int u) -> node {
        if(l <= tr[u].l && tr[u].r <= r) {
            return tr[u];
        }
        pushdown(u);
        if(r <= tr[ls].r) return query(query, l, r, ls);
        if(l >= tr[rs].l) return query(query, l, r, rs);
        return merge(query(query, l, r, ls), query(query, l, r, rs));
    };
    build(build, 1, n, 1);

    LL ans = 0;
    FOR(i, 1, q) {
        LL wei = a[i][0];
        int l = a[i][1], r = a[i][2];
        node x = query(query, l, r, 1);
        int prel = x.prel, prer = x.prer, cnt = x.ans;
        ans += cnt * wei;
        modify(modify, left[prel], right[prer], 1, prel);
        right[prel] = right[prer];
    }
    if(query(query, 1, n, 1).ans == 1) cout << ans << endl;
    else cout << -1 << endl;
}

signed main() {
    cin.tie(0), cout.tie(0) -> sync_with_stdio(0);
    int t = 1;
    // cin >> t;
    while(t--) solve();
    return 0;
}

analysis

这题是斯坦纳树的板子题。最小斯坦纳树问题指的是，给一个无向图，求出将其中 kkk 个点连接在一起的最小的代价。显然这个问题的答案一定是一个树，我们用状压来解决这道题。

设 dp(i, dp(i,\,dp(i, S)S)S) 表示的是 iii 为根且包含集合 SSS 中的点的树的代价的最小值，下面考虑状态转移。

  • 如果 dp(i,  S)dp(i,\;S)dp(i,S) 所对应的树的根（即 iii ）的度为 111，则有

dp(i,  S)=w(i,  j)+dp(j,  S)dp(i,\;S) = w(i,\;j) + dp(j,\;S) dp(i,S)=w(i,j)+dp(j,S)

  • 如果 dp(i,  S)dp(i,\;S)dp(i,S) 所对应的树的根的度大于 111，则有

dp(i,  S)=dp(i,  T)+dp(i,  S−T)dp(i,\;S) = dp(i,\;T) + dp(i,\;S - T) dp(i,S)=dp(i,T)+dp(i,S−T)

只需要枚举各种转移的情况即可求解 dp(i,  S)dp(i,\;S)dp(i,S)。对于下面的转移方程，实现起来是简单的，只需要按 SSS 从小到大的顺序求解 dp(i,  S)dp(i,\;S)dp(i,S) 即可。对于上面的转移方程，则可以使用最短路算法中的松弛操作来进行处理，这里采用 dijkstra\text{dijkstra}dijkstra 算法中的松弛操作来解决。

#include<bits/stdc++.h>
#define pii pair<LL, int>
#define endl "\n"
#define setpre(x) fixed << setprecision(x)
#define FOR(i, l, r) for(int i = l; i <= r; ++i)
#define ROF(i, l, r) for(int i = r; i >= l; --i)
#define fi first
#define se second
using namespace std;
using LL = long long;
constexpr int MAXN = 5e5 + 10;
constexpr int inf = 0x3f3f3f3f;
constexpr LL  INF = 0x3f3f3f3fffffffff;

struct graph {
    int n, m, cnt;
    vector<int> head, nxt, to;
    vector<LL> w;
    graph(int n, int m, int direct) : n(n), m(m) {
        if(!direct) m <<= 1;
        cnt = 0;
        head.resize(n + 1, -1);
        nxt.resize(m + 1);
        to.resize(m + 1);
        w.resize(m + 1);
    }
    void add(int u, int v, LL wei = 0) {
        nxt[cnt] = head[u];
        to[cnt] = v;
        w[cnt] = wei;
        head[u] = cnt;
        cnt++;
    }
};

void solve() {
    int n, m, k;
    cin >> n >> m >> k;
    --k;
    graph g(n, m, 0);
    FOR(i, 1, m) {
        int u, v, w;
        cin >> u >> v >> w;
        g.add(u, v, w), g.add(v, u, w);
    }
    
    vector<vector<LL>> dp(n + 1, vector<LL> (1 << k, INF));
    for(int i = 1; i <= k; ++i) dp[i][1 << (i - 1)] = 0;

    for(int s = 1; s < 1 << k; ++s) {
        priority_queue<pii, vector<pii>, greater<pii>> q;
        for(int i = 1; i <= n; ++i) {
            for(int t = (s - 1) & s; t; t = (t - 1) & s) {
                dp[i][s] = min(dp[i][s], dp[i][t] + dp[i][s ^ t]);
            }
            q.push({dp[i][s], i});
        }
        while(!q.empty()) {
            auto [dis, u] = q.top();
            q.pop();
            if(dis > dp[u][s]) continue;
            for(int i = g.head[u]; ~i; i = g.nxt[i]) {
                int v = g.to[i];
                LL w = g.w[i];
                if(dp[u][s] + w < dp[v][s]) {
                    dp[v][s] = dp[u][s] + w;
                    q.push({dp[v][s], v});
                }
            }
        }
    }

    for(int i = k + 1; i <= n; ++i) {
        cout << dp[i][(1 << k) - 1] << endl;
    }
}

signed main() {
    cin.tie(0), cout.tie(0) -> sync_with_stdio(0);
    int t = 1;
    // cin >> t;
    while(t--) solve();
    return 0;
}
